cmake_minimum_required(VERSION 3.8)
project(cartographer_parallel LANGUAGES CXX CUDA)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  std_msgs
  sensor_msgs
  geometry_msgs
  nav_msgs
  visualization_msgs
)

catkin_package(
  INCLUDE_DIRS include
  LIBRARIES assignment_cpu_lib fast_matcher_lib
  CATKIN_DEPENDS roscpp std_msgs sensor_msgs geometry_msgs nav_msgs visualization_msgs
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CUDA_STANDARD 14)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# Jetson Nano Developer Kit: CUDA compute capability 5.3
set(CMAKE_CUDA_FLAGS
    "${CMAKE_CUDA_FLAGS} -gencode arch=compute_53,code=sm_53")

# score_all_submit.cpp is compiled as CUDA so __CUDACC__ is defined
# and the GPU kernel is active.
add_library(assignment_cpu_lib
  src/score_all_submit.cpp
)
set_source_files_properties(src/score_all_submit.cpp PROPERTIES LANGUAGE CUDA)
target_link_libraries(assignment_cpu_lib ${catkin_LIBRARIES} cudart)

add_library(fast_matcher_lib
  src/fast_matcher.cpp
)
target_link_libraries(fast_matcher_lib assignment_cpu_lib ${catkin_LIBRARIES})

add_executable(fast_correlative_node src/fast_correlative_node.cpp)
target_link_libraries(fast_correlative_node fast_matcher_lib ${catkin_LIBRARIES})

# Multi-call profiler: warm-up aware, reports steady-state GPU time + correctness check.
add_executable(profile_score_all_multi src/profile_score_all_multi.cpp)
target_link_libraries(profile_score_all_multi assignment_cpu_lib ${catkin_LIBRARIES})

install(TARGETS
  assignment_cpu_lib
  fast_matcher_lib
  fast_correlative_node
  profile_score_all_multi
  ARCHIVE DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

install(DIRECTORY launch maps bags
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
